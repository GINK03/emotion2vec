(Constructor Injection) \u3068\u547c\u3070\u308c\u308b\u30c6\u30af\u30cb\u30c3\u30af\u3067\u3059\u3002\n\n\u3053\u306e\u5834\u5408\u306e\u30c6\u30b9\u30c8\u30b1\u30fc\u30b9\u306f\u4ee5\u4e0b\u306e\u3088\u3046\u306b\u306a\u308a\u307e\u3059\u3002\n\n```php\n    public function testPlay() {\n        // Bar \u30af\u30e9\u30b9\u306e\u30e2\u30c3\u30af\u3092\u4f5c\u6210\u3059\u308b\u3002\n        $bar = $this->getMock('Bar');\n        $bar->expects($this->any())\n            ->method('getSomething')\n            ->will($this->returnValue(1));\n\n        $foo = new Foo($bar);   // Constructor Injection \u3059\u308b\u3088\u3046\u306b\u3057\u3066\u3044\u307e\u3059\u3002\u3053\u308c\u306b\u3088\u308a\u3001\u30c6\u30b9\u30c8\u306e\u6642\u306f\u30e2\u30c3\u30af\u3092\u6ce8\u5165\u3057\u3066\u4f7f\u3046\u3088\u3046\u306b\u3057\u3001\u672c\u756a\u30b3\u30fc\u30c9\u3067\u306f\u30d1\u30e9\u30e1\u30fc\u30bf\u3092\u4f55\u3082\u6307\u5b9a\u3057\u306a\u3044\u3053\u3068\u3067\u30c7\u30d5\u30a9\u30eb\u30c8\u306e Foo \u30af\u30e9\u30b9\u3092\u4f7f\u3046\u3088\u3046\u306b\u3057\u3066\u3044\u307e\u3059\u3002\n\n\n\u30bb\u30c3\u30bf\u30fc\u30fb\u30a4\u30f3\u30b8\u30a7\u30af\u30b7\u30e7\u30f3\n--------------------------\n\u5225\u306e\u30a2\u30d7\u30ed\u30fc\u30c1\u3068\u3057\u3066\u3001\u30af\u30e9\u30b9\u306b\u30aa\u30d6\u30b8\u30a7\u30af\u30c8\u3092\u6ce8\u5165\u3059\u308b\u305f\u3081\u30bb\u30c3\u30bf\u30fc\u30e1\u30bd\u30c3\u30c9\u3092\u7528\u610f\u3059\u308b\u65b9\u6cd5\u3082\u3042\u308a\u307e\u3059\u3002\n\n```php\nclass 1) {\n            return true;\n        }\n\n        return false;\n    }\n}\n```\n\n\u30b3\u30f3\u30b9\u30c8\u30e9\u30af\u30bf\u30fc\u3067 Bar \u30aa\u30d6\u30b8\u30a7\u30af\u30c8\u3092\u6ce8\u5165\u3057\u3066\u3044\u307e\u3059\u3002\n\n\u3053\u306e\u5834\u5408\u306e\u30c6\u30b9\u30c8\u30b1\u30fc\u30b9\u306f\u4ee5\u4e0b\u306e\u3088\u3046\u306b\u306a\u308b\u3067\u3057\u3087\u3046\u3002\n\n```php\n
